1function FuelConsTable = fuel()
 2% Read fuel consumption data for Toyota Prius from 1999
 3fileSpeed = fopen('data/speed_rpm.txt','r');
 4fileTorque = fopen('data/torque_nm.txt','r');
 5fileFuel = fopen('data/fuel_gps.txt','r');
 6filePower = fopen('data/power_kw.txt','r');
 7coder.extrinsic('float');
 8coder.extrinsic('fit');
 9
10formatSpec = '%f';
11
12FuelCons = struct('speed_rpm',[], 'torque_Nm',[], 'fuel_gps',[], 'power_kw', []);
13FuelConsTable = struct('speed', [], 'torque', [], 'fuel', [], 'power', [], 'lookupTableFuel', zeros(10,10), 'lookupTablePower', zeros(10,10));
14coder.extrinsic('fscanf');
15FuelCons.speed_rpm = fscanf(fileSpeed,formatSpec);
16FuelCons.torque_Nm = fscanf(fileTorque,formatSpec);
17FuelCons.power_kw = fscanf(filePower,formatSpec);
18fuel_galps = fscanf(fileFuel,formatSpec);
19
20gasoline_grams_per_m3 = 719.7 * 1000;    % density 719.7 kg/m3 * 1000 grams
21gallons_per_m3 = 264.172;
22
23FuelCons.fuel_gps = (fuel_galps / gallons_per_m3) * gasoline_grams_per_m3;
24
25num_points_fuel = 10;
26FuelConsTable.speed = linspace(float(min(FuelCons.speed_rpm)), max(FuelCons.speed_rpm), num_points_fuel);
27FuelConsTable.torque = linspace(float(min(FuelCons.torque_Nm)), max(FuelCons.torque_Nm), num_points_fuel);
28FuelConsTable.fuel = linspace(float(min(FuelCons.fuel_gps)), max(FuelCons.fuel_gps), num_points_fuel);
29FuelConsTable.power = linspace(float(min(FuelCons.power_kw)), max(FuelCons.power_kw), num_points_fuel);
30
31speedMeshgridFuel = zeros(size(FuelConsTable.speed,2), size(FuelConsTable.torque,2));
32torqueMeshgridFuel = zeros(size(FuelConsTable.speed,2), size(FuelConsTable.torque,2));
33
34[speedMeshgridFuel, torqueMeshgridFuel] = meshgrid(FuelConsTable.speed, FuelConsTable.torque);
35x = horzcat(FuelConsTable.speed(:), FuelConsTable.torque(:));
36
37fuelFit = fit([FuelConsTable.speed(:), FuelConsTable.torque(:)], FuelConsTable.fuel(:), 'poly22', 'Normalize', 'on');
38powerFit = fit([FuelConsTable.speed(:), FuelConsTable.torque(:)], FuelConsTable.power(:), 'poly23', 'Normalize', 'on');
39%FuelConsTable.lookupTableFuel = zeros(size(FuelConsTable.speed,2), size(FuelConsTable.torque,2));
40%FuelConsTable.lookupTablePower = zeros(size(FuelConsTable.speed,2), size(FuelConsTable.torque,2));
41%Xhat = feval(fuelFit, FuelConsTable.fuel(:));
42
43FuelConsTable.lookupTableFuel = feval(fuelFit,speedMeshgridFuel,torqueMeshgridFuel)
44FuelConsTable.lookupTablePower = feval(powerFit,speedMeshgridFuel,torqueMeshgridFuel)
45
46
47
48
49figure
50plot3(FuelCons.speed_rpm, FuelCons.torque_Nm, FuelCons.fuel_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
51grid on
52box on
53xlabel('Engine Speed (rpm)','FontSize',11)
54ylabel('Engine Torque (Nm)','FontSize',11)
55zlabel('Fuel Rate (g/s)','FontSize',11)
56title('Fuel Consumption','FontSize',13,'FontWeight','Bold')
57set(gca,'XLim',[1200 4100]);
58set(gca,'YLim',[20 110]);
59set(gca,'ZLim',[0 2.85]);
60
61figure
62plot3(FuelCons.speed_rpm, FuelCons.torque_Nm, FuelCons.power_kw,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
63grid on
64box on
65xlabel('Engine Speed (rpm)','FontSize',11)
66ylabel('Engine Torque (Nm)','FontSize',11)
67zlabel('Power (kW)','FontSize',11)
68title('Power Consumption','FontSize',13,'FontWeight','Bold')
69set(gca,'XLim',[1200 4100]);
70set(gca,'YLim',[20 110]);
71set(gca,'ZLim',[0 45]);
72end
