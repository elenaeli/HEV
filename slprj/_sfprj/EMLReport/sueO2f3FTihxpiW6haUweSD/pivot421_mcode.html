 2function B = pivot(A,r,s)
 3% Pivots the tableau on the given row and column
 4    
 5    [m,~] = size(A);
 6    B = A;
 7    
 8    for i = 1 : m
 9        if i == r
10            continue;
11        else
12            B(i,:) = A(i,:) - A(i,s) / A(r,s) * A(r,:);
13        end
14    end
15    
16end
