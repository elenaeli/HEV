1function exhaustEmissions()
 2fileSpeed = fopen('data/speed_rpm.txt','r');
 3fileTorque = fopen('data/torque_nm.txt','r');
 4
 5fileHC = fopen('data/hc_gps.txt','r');
 6fileCO = fopen('data/co_gps.txt','r');
 7fileNOX = fopen('data/nox_gps.txt','r');
 8formatSpec = '%f';
 9coder.extrinsic('fscanf')
10
11GasEmis = struct('speed_rpm',[], 'torque_Nm',[], 'hc_gps',[], 'co_gps',[], 'nox_gps',[]);
12GasEmis.speed_rpm = fscanf(fileSpeed,formatSpec);
13GasEmis.torque_Nm = fscanf(fileTorque,formatSpec);
14GasEmis.hc_gps = fscanf(fileHC,formatSpec);
15GasEmis.co_gps = fscanf(fileCO,formatSpec);
16GasEmis.nox_gps = fscanf(fileNOX,formatSpec);
17
18num_points_gas = 10;
19GasEmisTable.speed = linspace(min(GasEmis.speed_rpm), max(GasEmis.speed_rpm), num_points_gas);
20GasEmisTable.torque = linspace(min(GasEmis.torque_Nm), max(GasEmis.torque_Nm), num_points_gas);
21GasEmisTable.hc = linspace(min(GasEmis.hc_gps), max(GasEmis.hc_gps), num_points_gas);
22GasEmisTable.nox = linspace(min(GasEmis.nox_gps), max(GasEmis.nox_gps), num_points_gas);
23GasEmisTable.co = linspace(min(GasEmis.co_gps), max(GasEmis.co_gps), num_points_gas);
24
25[speedMeshgridGas, torqueMeshgridGas] = meshgrid(GasEmisTable.speed, GasEmisTable.torque);
26gasFitHC = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.hc(:), 'poly22');
27GasEmisTable.lookupTableHC = gasFitHC(speedMeshgridGas, torqueMeshgridGas);
28
29gasFitCO = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.co(:), 'poly22');
30GasEmisTable.lookupTableCO = gasFitCO(speedMeshgridGas, torqueMeshgridGas);
31
32gasFitNOX = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.nox(:), 'poly22');
33GasEmisTable.lookupTableNOX = gasFitNOX(speedMeshgridGas, torqueMeshgridGas);
34close all
35
36figure
37plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.hc_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
38grid on
39box on
40xlabel('Engine Speed (rpm)','FontSize',11)
41ylabel('Engine Torque (Nm)','FontSize',11)
42zlabel('HC Emissions (g/s)','FontSize',11)
43title('Hydrogen Carbon Emissions','FontSize',13,'FontWeight','Bold')
44set(gca,'XLim',[1200 4100]);
45set(gca,'YLim',[20 110]);
46set(gca,'ZLim',[0 0.06]);
47
48figure
49plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.co_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
50grid on
51box on
52xlabel('Engine Speed (rpm)','FontSize',11)
53ylabel('Engine Torque (Nm)','FontSize',11)
54zlabel('CO Emissions (g/s)','FontSize',11)
55title('Carbon Oxide Emissions','FontSize',13,'FontWeight','Bold')
56set(gca,'XLim',[1200 4100]);
57set(gca,'YLim',[20 110]);
58set(gca,'ZLim',[0 0.22]);
59
60figure
61plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.nox_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
62grid on
63box on
64xlabel('Engine Speed (rpm)','FontSize',11)
65ylabel('Engine Torque (Nm)','FontSize',11)
66zlabel('NOX Emissions (g/s)','FontSize',11)
67title('Nitrogen Oxide Emissions','FontSize',13,'FontWeight','Bold')
68set(gca,'XLim',[1200 4100]);
69set(gca,'YLim',[20 110]);
70set(gca,'ZLim',[0 0.16]);
71end
