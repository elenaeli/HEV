1function GasEmisTable = exhaustEmissions()
 2fileSpeed = fopen('data/speed_rpm.txt','r');
 3fileTorque = fopen('data/torque_nm.txt','r');
 4
 5fileHC = fopen('data/hc_gps.txt','r');
 6fileCO = fopen('data/co_gps.txt','r');
 7fileNOX = fopen('data/nox_gps.txt','r');
 8formatSpec = '%f';
 9coder.extrinsic('fscanf')
10coder.extrinsic('fit')
11coder.extrinsic('float')
12
13GasEmis = struct('speed_rpm',[], 'torque_Nm',[], 'hc_gps',[], 'co_gps',[], 'nox_gps',[]);
14GasEmisTable = struct ('speed', [], 'torque', [], 'hc', [], 'nox', [], 'co', [], 'lookupTableHC', zeros(10,10), 'lookupTableCO', zeros(10,10), 'lookupTableNOX', zeros(10,10));
15GasEmis.speed_rpm = fscanf(fileSpeed,formatSpec);
16GasEmis.torque_Nm = fscanf(fileTorque,formatSpec);
17GasEmis.hc_gps = fscanf(fileHC,formatSpec);
18GasEmis.co_gps = fscanf(fileCO,formatSpec);
19GasEmis.nox_gps = fscanf(fileNOX,formatSpec);
20
21num_points_gas = 10;
22GasEmisTable.speed = linspace(float(min(GasEmis.speed_rpm)), max(GasEmis.speed_rpm), num_points_gas);
23GasEmisTable.torque = linspace(float(min(GasEmis.torque_Nm)), max(GasEmis.torque_Nm), num_points_gas);
24GasEmisTable.hc = linspace(float(min(GasEmis.hc_gps)), max(GasEmis.hc_gps), num_points_gas);
25GasEmisTable.nox = linspace(float(min(GasEmis.nox_gps)), max(GasEmis.nox_gps), num_points_gas);
26GasEmisTable.co = linspace(float(min(GasEmis.co_gps)), max(GasEmis.co_gps), num_points_gas);
27
28[speedMeshgridGas, torqueMeshgridGas] = meshgrid(GasEmisTable.speed, GasEmisTable.torque);
29gasFitHC = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.hc(:), 'poly22');
30GasEmisTable.lookupTableHC = feval(gasFitHC,speedMeshgridGas,torqueMeshgridGas);
31
32gasFitCO = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.co(:), 'poly22');
33GasEmisTable.lookupTableCO = feval(gasFitCO,speedMeshgridGas,torqueMeshgridGas);
34
35gasFitNOX = fit([GasEmisTable.speed(:), GasEmisTable.torque(:)], GasEmisTable.nox(:), 'poly22');
36GasEmisTable.lookupTableNOX = feval(gasFitNOX,speedMeshgridGas,torqueMeshgridGas);
37close all
38
39figure
40plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.hc_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
41grid on
42box on
43xlabel('Engine Speed (rpm)','FontSize',11)
44ylabel('Engine Torque (Nm)','FontSize',11)
45zlabel('HC Emissions (g/s)','FontSize',11)
46title('Hydrogen Carbon Emissions','FontSize',13,'FontWeight','Bold')
47set(gca,'XLim',[1200 4100]);
48set(gca,'YLim',[20 110]);
49set(gca,'ZLim',[0 0.06]);
50
51figure
52plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.co_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
53grid on
54box on
55xlabel('Engine Speed (rpm)','FontSize',11)
56ylabel('Engine Torque (Nm)','FontSize',11)
57zlabel('CO Emissions (g/s)','FontSize',11)
58title('Carbon Oxide Emissions','FontSize',13,'FontWeight','Bold')
59set(gca,'XLim',[1200 4100]);
60set(gca,'YLim',[20 110]);
61set(gca,'ZLim',[0 0.22]);
62
63figure
64plot3(GasEmis.speed_rpm, GasEmis.torque_Nm, GasEmis.nox_gps,'o','MarkerFaceColor','blue','MarkerEdgeColor','white');
65grid on
66box on
67xlabel('Engine Speed (rpm)','FontSize',11)
68ylabel('Engine Torque (Nm)','FontSize',11)
69zlabel('NOX Emissions (g/s)','FontSize',11)
70title('Nitrogen Oxide Emissions','FontSize',13,'FontWeight','Bold')
71set(gca,'XLim',[1200 4100]);
72set(gca,'YLim',[20 110]);
73set(gca,'ZLim',[0 0.16]);
74end
