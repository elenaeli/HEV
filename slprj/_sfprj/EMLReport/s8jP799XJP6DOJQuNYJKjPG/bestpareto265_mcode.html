1function [ bestPayoffEngPareto, bestPayoffMotPareto ] = bestpareto( paretoStrategies, paretoIndex, ...
 2    torqueDeviation, fuel, nox, power )
 3    payoffEngPareto = zeros(1,size(paretoIndex,1));
 4    payoffMotPareto = zeros(1,size(paretoIndex,1));
 5    torqueDeviationPareto = zeros(1,size(paretoIndex,1));
 6    fuelPareto = zeros(1,size(paretoIndex,1));
 7    noxPareto = zeros(1,size(paretoIndex,1));
 8    powerPareto = zeros(1,size(paretoIndex,1));
 9    for r = 1 : size(paretoIndex,1)
10        payoffEngPareto(1,r) = paretoStrategies(r,1);
11        payoffMotPareto(1,r) = paretoStrategies(r,2);
12        torqueDeviationPareto(1,r) = torqueDeviation(paretoIndex(r,1), paretoIndex(r,2));
13        fuelPareto(1,r) = fuel(paretoIndex(r,1),1);
14        noxPareto(1,r) = nox(paretoIndex(r,1),1);
15        powerPareto(1,r) = power(paretoIndex(r,1),1);
16    end        
17       
18    bestTorqDevPareto = min(abs(torqueDeviationPareto));
19       
20    % check if more than 1 points have same torque deviation
21    indT = find(abs(torqueDeviationPareto)==bestTorqDevPareto);   
22    if size(indT,2) > 1
23        % take the point with minimum fuel consumption
24        bestFuelPareto = min(fuelPareto);
25        % check if more than 1 points have same fuel consumption
26        indF = find(fuelPareto==bestFuelPareto);
27        bestPayoffFuelPareto = paretoStrategies(indF,:);
28        if size(indF,2) > 1
29            % take the point with the minimum engine payoff
30            [~,indE] = min(powerPareto(indF));
31            bestPayoffEngPareto = bestPayoffFuelPareto(indE,1); 
32            bestPayoffMotPareto = bestPayoffFuelPareto(indE,2);
33        else
34            bestPayoffEngPareto = bestPayoffFuelPareto(1,1);   
35            bestPayoffMotPareto = bestPayoffFuelPareto(1,2);
36        end
37    else
38        bestPayoffEngPareto = paretoStrategies(indT,1);
39        bestPayoffMotPareto = paretoStrategies(indT,2);
40    end    
41end
