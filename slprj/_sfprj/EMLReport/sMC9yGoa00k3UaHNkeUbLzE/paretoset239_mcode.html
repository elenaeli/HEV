1function  [paretoStrategies, paretoIndices, x, y]  = paretoset( payoffEngine, payoffMotor)
 2    coder.extrinsic('legend');
 3    coder.extrinsic('int2str');
 4    coder.extrinsic('ismembertol');
 5   
 6    [k,l] = size(payoffEngine);    
 7    Eng = payoffEngine;
 8    Mot = payoffMotor;
 9
10    x = reshape(Eng',1,k*l);
11    y = reshape(Mot',1,k*l);
12    
13    paretoIndices = zeros(k,2);
14    paretoStrategies = zeros(k,2);
15    coder.varsize('paretoIndex');
16    coder.varsize('paretoStrategies');
17    coder.varsize('paretoIndices');
18    count = 1;
19    
20    for i=1:k      
21        for j=1:l                
22            isNotPareto = 0;
23            flag = 0;  
24            for m=1:k                  
25                for n=1:l
26                    if Eng(i,j) > Eng(m,n) && Mot(i,j) > Mot(m,n)                    
27                        isNotPareto = 1;
28                        flag = 1;
29                        break;
30                    end                      
31                end
32                if flag == 1
33                    break;
34                end
35            end
36            if isNotPareto == 0                  
37                repeatedRow = ismembertol(paretoStrategies,[Eng(i,j) Mot(i,j)], 0.000001, 'ByRows', true);
38                if ~any(repeatedRow)    
39                    paretoIndices(count,1) = i;
40                    paretoIndices(count,2) = j;
41                    paretoStrategies(count,1) = Eng(i,j);
42                    paretoStrategies(count,2) = Mot(i,j);
43                    count = count+1;
44                end
45            end           
46        end
47    end
48    
49    %[pareto,ia,~] = unique(pareto,'rows');   
50    %paretoIndex = paretoIndex(ia, :); 
51    
52    % remove all rows with zeros    
53
54    paretoIndices( ~any(paretoIndices,2), : ) = [];
55    paretoStrategies( ~any(paretoStrategies,2), : ) = [];
56   
57end
