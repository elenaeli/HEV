 3    paretoString = [];

 7
 8    ind = 1;

11

20    end
21

24    y = [y y(1)];
25
26    count = 1;
27    paretoIndex = zeros(1,2);

30        for j=1:l          
31            not = 0;
32            flag = 0;
33            for m=1:k  
34                for n=1:l                
35                    if E(i,j) > E(m,n) && M(i,j) > M(m,n)                    
36                        not = 1;
37                        flag = 1;
38                        break;
39                    end                      
40                end
41                if flag == 1
42                    break;
43                end
44            end            
45            if not == 0            
46                paretoIndex(count,1) = i;
47                paretoIndex(count,2) = j;
48                pareto(count,1) = M(i,j);
49                pareto(count,2) = E(i,j);
50                count = count+1;
51            end            
52        end
53    end

65        linearIndex = sub2ind([k l], paretoIndex(i,2), paretoIndex(i,1));
66        plot(x(linearIndex), y(linearIndex), 'or', 'MarkerFaceColor','r');   
67        coder.extrinsic('legend');
68        legend('payoff','Pareto optimum payoff', 'Location','northeast');
69        xlabel('Payoff Motor');
70        ylabel('Payoff Engine');
71        pareto(i,:) = [paretoIndex(i,1), paretoIndex(i,2)];
72        paretoString(i) = strcat('(M',int2str(paretoIndex(i,1)),',E',int2str(paretoIndex(i,2)),')');
