1function [nashEqbm1, nashEqbm2] = LemkeHowson(varargin)
  2    % function nashEqbm = LEMKEHOWSON(varargin)
  3    %
  4    % This function computes a sample mixed strategy Nash equilibrium in a
  5    % bimatrix game.  This function implements the Lemke-Howson complementary
  6    % pivoting algorithm for solving Bimatrix Games, a variant of the Lemke 
  7    % algorithm for linear complementarity problems (LCPs).
  8    %
  9    % Syntax:
 10    %   nashEqbm = LEMKEHOWSON(A, B)
 11    %   nashEqbm = LEMKEHOWSON(A, B, k0)
 12    %   nashEqbm = LEMKEHOWSON(A, B, k0, maxPivots)
 13    %
 14    % Parameters:
 15    %   A an m*n payoff matrix for the row player
 16    %   B an m*n payoff matrix for the column player
 17    %   k0 an initial pivot in the set {1,...,m+n}
 18    %       (optional: default = 1)
 19    %   maxPivots the maximum number of pivoting steps before termination
 20    %       (optional: default = 500000);
 21    %
 22    % Return:
 23    %   nashEqbm a 2x1 cell array where nashEqbm{1} and nashEqbm{2} are mixed
 24    %   strategies for the row and column player, respectively.
 25    %
 26    % Author:
 27    %   Richard M. Katzwer
 28    %   9/18/2013
 29    %   princeton.edu/~rkatzwer
 30    %
 31    % Updated:
 32    %   11/13/2013
 33    % 
 34    % References:
 35    %    C. E. Lemke and J. T. Howson, Jr.
 36    %      "Equilibrium Points of Bimatrix Games"
 37    %      Journal of the Society for Industrial and Applied Mathematics.
 38    %      Vol. 12, No. 2 (Jun., 1964), pp. 413-423
 39    %    Lloyd S. Shapley. "A note on the Lemke-Howson algorithm".
 40    %      Pivoting and Extension: Mathematical Programming Studies
 41    %      Volume 1, 1974, pp 175-189
 42    %    Bruno Codenotti, Stefano De Rossi, Marino Pagan.
 43    %      "An experimental analysis of Lemke-Howson algorithm."
 44    
 45    %% Check inputs
 46     coder.extrinsic('num2str');
 47     nashEqbm1 = 0;
 48     nashEqbm2 = 0;
 49     if nargin < 2 || nargin > 4
 50        error('This function takes between two and four arguments');
 51    end
 52    
 53    A = varargin{1};
 54    B = varargin{2};
 55    
 56    if any(size(A) ~= size(B))
 57        error('Matrices must have same dimension');
 58    end
 59    
 60    [m,n] = size(A);
 61    size_ = [m,n];
 62    
 63    if nargin > 2
 64        k0 = varargin{3};
 65        if k0 < 1 || k0 > m+n
 66            error(['Initial pivot must be in {1,...,' num2str(n+m) '}']);
 67        end
 68    else
 69        k0 = 1;
 70    end
 71    
 72    if nargin == 4
 73        maxPivots = varargin{4};
 74        if maxPivots < 1
 75            error('Maximum pivots parameter must be a positive integer!');
 76        end
 77    else
 78        maxPivots = 500000;
 79    end
 80    
 81    %% Scale payoffs to be strictly positive
 82    minVal = min( min(min(A)), min(min(B)) );
 83    if minVal <= 0
 84        A = A + ones(size(A))*(1-minVal);
 85        B = B + ones(size(A))*(1-minVal);
 86    end
 87    
 88    %% Build Tableaus
 89    %Tab = cell(2,1);
 90    Tab1 = [B',     eye(n), ones(n,1)];
 91    Tab2 = [eye(m), A,      ones(m,1)];
 92    
 93    %% Declare row labels
 94    %rowLabels = cell(2,1);
 95    rowLabels1 = m+1:m+n;
 96    rowLabels2 = 1:m;
 97    
 98    %% Do complementary pivoting
 99    k = k0;
100    if k0 <= m
101        player = 1;
102    else
103        player = 2;
104    end
105    
106    % Pivoting loop
107    numPiv = 0;
108    while numPiv < maxPivots
109        
110        numPiv = numPiv+1;        
111        % Use correct Tableau
112        if player == 1
113            LP = Tab1; 
114        else
115            LP = Tab2;
116        end
117      
118        [m_, ~] = size(LP);
119        
120        % Find pivot row (variable exiting)
121        max_ = 0;
122        ind = -1;
123        for i = 1:m_
124            t = LP(i,k) / LP(i, m+n+1);
125            if t > max_
126                ind = i;
127                max_ = t;
128            end
129        end
130        
131        if max_ > 0
132             if player == 1
133                Tab1 = pivot(LP, ind, k);
134             else
135                Tab2 = pivot(LP, ind, k);
136             end         
137        else
138            break;
139        end
140        
141        % swap labels, set entering variable
142         if player == 1
143              temp = rowLabels1(ind);
144              rowLabels1(ind) = k;
145         else
146             temp = rowLabels2(ind);
147              rowLabels2(ind) = k;
148         end
149       
150        k = temp;
151        
152        % If the entering variable is the same
153        % as the starting pivot, break
154        if k == k0
155            break;
156        end
157        
158        % update the tableau index
159        if player == 1
160            player = 2;
161        else
162            player = 1;
163        end
164        
165    end
166   
167    if numPiv == maxPivots
168        error(['Maximum pivot steps (' num2str(maxPivots) ') reached!']);
169    end
170    
171    %% Extract the Nash equilibrium
172       
173    for player = 1:2
174        
175        x = zeros(size_(player), 1);
176        if player == 1
177            rows = rowLabels1;
178            LP = Tab1;
179        else
180            rows = rowLabels2;
181            LP = Tab2;
182        end
183        
184        for i = 1:length(rows)
185            if player == 1 && rows(i) <= size_(1)
186                x(rows(i)) = LP(i,m+n+1) / LP(i,rows(i));
187            elseif player == 2 && rows(i) > size_(1);
188                x(rows(i)-size_(1)) = LP(i,m+n+1) / LP(i,rows(i));
189            end
190        end
191        
192        if player == 1
193            nashEqbm1= x/sum(x);
194        else
195            nashEqbm2= x/sum(x);
196        end
197    end
198end
