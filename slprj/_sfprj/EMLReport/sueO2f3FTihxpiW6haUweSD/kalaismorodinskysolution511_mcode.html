1function [ ks, linePareto] = kalaismorodinskysolution( payoffBoth, conflictPoint, pareto, paretoStrInd )    
  2    % A function to find the Kalai Smorodinsky bargaining solution. It
  3    % is the intersection of the line between the ideal point where both 
  4    % players get minimum payoffs and the conflict point (the same as in 
  5    % Nash Solution. The conflict point is taken to be the Nash 
  6    % Equilibrium) with the Pareto Front.
  7    
  8    % ideal point m (often not feasible)
  9    m1 = min(payoffBoth(:,1));
 10    m2 = min(payoffBoth(:,2));    
 11    %darkred = [185 16 20] ./ 255;
 12    %plot(m1, m2, 'o', 'MarkerSize', 3, 'MarkerFaceColor',darkred ,...
 13    %    'MarkerEdgeColor', darkred);
 14   
 15    % conflict and ideal point
 16    cP = [payoffBoth(conflictPoint,1) payoffBoth(conflictPoint,2)];
 17    mP = [m1 m2];
 18    xCM = [cP(1) mP(1)];
 19    yCM = [cP(2) mP(2)];
 20   
 21    % j = 1 for different x and same y
 22    % j = 2 for different y and same x (for the ideal and conflict points)
 23    if mP(2) == cP(2)
 24        j = 1;
 25    elseif mP(1) == cP(1)
 26        j = 2;
 27    else
 28        j = 1;        
 29    end
 30   
 31    % interpolate to get points between ideal and conflict point
 32    if mP(j) < cP(j) 
 33        X = linspace(mP(j), cP(j), 5);
 34        interpolate(j, xCM, yCM, X);
 35    elseif mP(j) > cP(j)
 36        X = linspace(cP(j), mP(j), 5);  
 37        interpolate(j, xCM, yCM, X);
 38    end
 39    % sort pareto points according to first column (X vaules)   
 40    paretoStrInd = sortrows(paretoStrInd,1);
 41    pareto = paretoStrInd(:,1:2);
 42    xPar = zeros(1,size(pareto,1));
 43    yPar = zeros(1,size(pareto,1));
 44    % get x and y coordinates of all pareto points
 45    for i = 1: size(pareto,1)       
 46        xPar(i) = pareto(i,1);
 47        yPar(i) = pareto(i,2);       
 48    end
 49    coder.varsize('payoffParetoFront', [500,5]);
 50    
 51    % according to the number of pareto points
 52    if size(pareto,1) == 1
 53        payoffParetoFront = horzcat(xPar, yPar);
 54    else
 55        % interpolate from smaller to bigger x coordinate (1st to 2nd
 56        % pareto point)      
 57        xP12 = linspace(pareto(1,1),pareto(2,1),100);       
 58        yP12 = interp1(xPar(1:2), yPar(1:2), xP12);
 59        plot(xP12,yP12,'b');      
 60        payoffParetoFront = horzcat(xP12', yP12');
 61        if size(pareto,1) >= 3
 62            % interpolate from smaller to bigger x coordinate (2nd to 3rd
 63            % pareto point)         
 64            xP23 = linspace(pareto(2,1),pareto(3,1),100);           
 65            yP23 = interp1(xPar(2:3), yPar(2:3), xP23);
 66            plot(xP23, yP23,'b');           
 67            payoffParetoFront = horzcat(vertcat(xP12',xP23'), vertcat(yP12',yP23'));
 68            if size(pareto,1) >= 4             
 69                xP34 = linspace(pareto(3,1),pareto(4,1),100);               
 70                yP34 = interp1(xPar(3:4), yPar(3:4), xP34);
 71                plot(xP34, yP34,'b');     
 72                payoffParetoFront = horzcat(vertcat(xP12',xP23',xP34'), vertcat(yP12',yP23',yP34'));
 73                if size(pareto,1) == 5            
 74                    xP45 = linspace(pareto(4,1),pareto(5,1),100);                                   
 75                    yP45 = interp1(xPar(4:5), yPar(4:5), xP45);
 76                    plot(xP45, yP45,'b');
 77                    payoffParetoFront = horzcat(vertcat(xP12',xP23',xP34',xP45'), vertcat(yP12',yP23',yP34',yP45'));
 78                end
 79            end         
 80        end
 81    end
 82            
 83    ks = zeros(1,2);   
 84    r = zeros(1,size(payoffParetoFront,1));
 85    l = zeros(1,size(payoffParetoFront,1));  
 86    
 87    for u = 1 : size(payoffParetoFront,1)       
 88        u2c2 = -payoffParetoFront(u,2) + payoffBoth(conflictPoint,2);
 89        u1c1 = -payoffParetoFront(u,1) + payoffBoth(conflictPoint,1);
 90        m2c2 = -m2 + payoffBoth(conflictPoint,2);
 91        m1c1 = -m1 + payoffBoth(conflictPoint,1);
 92        % left and right side of the KS equation definition
 93        l(u) = u2c2 / u1c1;
 94        r(u) = m2c2 / m1c1;
 95        
 96        % if either the numerator or denominator is 0, make the whole side
 97        % 0 to avoid NaN if denominator is 0.
 98        if all(u2c2 == 0) || all(u1c1 == 0)
 99            l(u) = 0;              
100        end
101        if all(m2c2 == 0) || all( m1c1 == 0)
102            r(u) = 0;
103        end        
104    end    
105    diff = r - l;
106    % take the minimum absolute difference to be the KS solution
107    [minDiff, minInd] = min(abs(diff));
108    ks(1,1) = payoffParetoFront(minInd,1);
109    ks(1,2) = payoffParetoFront(minInd,2); 
110    
111    linePareto = zeros(2,4);
112    if size(paretoStrInd,1) == 1
113         linePareto = paretoStrInd;
114    else
115        for i = 1: size(paretoStrInd,1)-1
116            if ks(1,1) >= paretoStrInd(i,1) && ks(1,1) <= paretoStrInd(i+1,1)
117                linePareto(1,:) = paretoStrInd(i,:);
118                linePareto(2,:) = paretoStrInd(i+1,:);
119            end
120        end
121    end
122end
