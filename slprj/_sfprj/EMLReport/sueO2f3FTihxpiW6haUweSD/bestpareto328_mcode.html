1function [ bestPayoffEngPareto, bestPayoffMotPareto ] = bestpareto( paretoStrategies, paretoIndex, ...
 2    torqueDeviation, fuel, nox, power )
 3    payoffEngPareto = zeros(1,size(paretoIndex,1));
 4    payoffMotPareto = zeros(1,size(paretoIndex,1));
 5    torqueDeviationPareto = zeros(1,size(paretoIndex,1));
 6    fuelPareto = zeros(1,size(paretoIndex,1));
 7    noxPareto = zeros(1,size(paretoIndex,1));
 8    powerPareto = zeros(1,size(paretoIndex,1));
 9    for r = 1 : size(paretoIndex,1)
10        payoffEngPareto(1,r) = paretoStrategies(r,1);
11        payoffMotPareto(1,r) = paretoStrategies(r,2);
12        torqueDeviationPareto(1,r) = torqueDeviation(paretoIndex(r,1), paretoIndex(r,2));
13        fuelPareto(1,r) = fuel(paretoIndex(r,1),1);
14        noxPareto(1,r) = nox(paretoIndex(r,1),1);
15        powerPareto(1,r) = power(paretoIndex(r,1),1);
16    end
17        
18    payoffEngPareto
19    payoffMotPareto        
20    bestTorqDevPareto = min(abs(torqueDeviationPareto))
21       
22    % check if more than 1 points have same torque deviation
23    indT = find(abs(torqueDeviationPareto)==bestTorqDevPareto);   
24    if size(indT,2) > 1
25        % take the point with minimum fuel consumption
26        bestFuelPareto = min(fuelPareto);
27        % check if more than 1 points have same fuel consumption
28        indF = find(fuelPareto==bestFuelPareto);
29        bestPayoffFuelPareto = paretoStrategies(indF,:);
30        if size(indF,2) > 1
31            % take the point with the minimum engine payoff
32            [~,indE] = min(powerPareto(indF));
33            bestPayoffEngPareto = bestPayoffFuelPareto(indE,1); 
34            bestPayoffMotPareto = bestPayoffFuelPareto(indE,2);
35        else
36            bestPayoffEngPareto = bestPayoffFuelPareto(1,1);   
37            bestPayoffMotPareto = bestPayoffFuelPareto(1,2);
38        end
39    else
40        bestPayoffEngPareto = paretoStrategies(indT,1);
41        bestPayoffMotPareto = paretoStrategies(indT,2);
42    end    
43end
